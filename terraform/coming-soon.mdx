## Coming Soon

The LibOps Terraform Provider is currently in development and will be available soon.

### What to Expect

The Terraform Provider will enable you to manage your entire LibOps infrastructure as code:

```hcl
# Example of what's coming
terraform {
  required_providers {
    libops = {
      source  = "libops/libops"
      version = "~> 1.0"
    }
  }
}

provider "libops" {
  api_url = "https://api.libops.io"
}

resource "libops_organization" "main" {
  name        = "ACME Corporation"
  description = "Main production infrastructure"
}

resource "libops_project" "web" {
  organization_id = libops_organization.main.id
  name            = "Web Applications"
  description     = "Customer-facing web apps"
}

resource "libops_site" "production" {
  project_id  = libops_project.web.id
  name        = "production-web"
  domain      = "www.acme.com"
  description = "Production website"
}

resource "libops_secret" "database_url" {
  site_id = libops_site.production.id
  key     = "DATABASE_URL"
  value   = var.database_url
}

resource "libops_firewall_rule" "office" {
  site_id     = libops_site.production.id
  cidr        = "203.0.113.0/24"
  description = "Office network"
}

resource "libops_member" "admin" {
  site_id = libops_site.production.id
  email   = "admin@acme.com"
  role    = "admin"
}
```

### Current Alternatives

#### Using the API with Terraform

You can use Terraform's `http` provider to interact with the LibOps API today:

```hcl
terraform {
  required_providers {
    http = {
      source = "hashicorp/http"
      version = "~> 3.0"
    }
  }
}

locals {
  api_url = "https://api.libops.io"
  api_token = var.libops_api_token
}

# Create an organization
resource "null_resource" "organization" {
  provisioner "local-exec" {
    command = <<-EOT
      curl -X POST ${local.api_url}/v1/organizations \
        -H "Authorization: Bearer ${local.api_token}" \
        -H "Content-Type: application/json" \
        -d '{"name":"ACME Corp","description":"Main infrastructure"}'
    EOT
  }
}
```

#### Using sitectl in Terraform

You can also call sitectl from Terraform:

```hcl
resource "null_resource" "create_site" {
  provisioner "local-exec" {
    command = <<-EOT
      sitectl create site \
        --project-id ${var.project_id} \
        --name production-web \
        --domain www.acme.com \
        --format json
    EOT
  }
}
```

<Note>
These are temporary workarounds. The official Terraform Provider will provide proper resource management, state tracking, and drift detection.
</Note>
